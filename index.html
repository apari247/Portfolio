const TelegramBot = require("node-telegram-bot-api"); // Replace with your token from @BotFather const TOKEN = process.env.TELEGRAM_BOT_TOKEN || "YOUR_BOT_TOKEN_HERE"; const bot = new TelegramBot(TOKEN, { polling: true }); console.log("🤖 Bot is running..."); // /start command bot.onText(/\/start/, (msg) => { const chatId = msg.chat.id; const firstName = msg.from.first_name || "there"; bot.sendMessage( chatId, `👋 Hello, ${firstName}! Welcome!\n\nI'm your bot. Use /help to see what I can do.` ); }); // /help command bot.onText(/\/help/, (msg) => { const chatId = msg.chat.id; bot.sendMessage( chatId, `📋 *Available Commands*\n\n` + `/start – Start the bot and get a welcome message\n` + `/help – Show this list of commands\n` + `/about – Learn more about this bot`, { parse_mode: "Markdown" } ); }); // /about command bot.onText(/\/about/, (msg) => { const chatId = msg.chat.id; bot.sendMessage( chatId, `ℹ️ *About This Bot*\n\n` + `This bot was built with Node.js using the \`node-telegram-bot-api\` library.\n\n` + `Version: 1.0.0`, { parse_mode: "Markdown" } ); }); // Handle unknown commands bot.on("message", (msg) => { const chatId = msg.chat.id; const text = msg.text || ""; if (text.startsWith("/") && !["start", "help", "about"].includes(text.slice(1).split(" ")[0])) { bot.sendMessage(chatId, `❓ Unknown command. Use /help to see available commands.`); } }); // Error handling bot.on("polling_error", (err) => { console.error("Polling error:", err.message); });
